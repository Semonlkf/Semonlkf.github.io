八皇后问题介绍

八皇后问题，是一个古老而著名的问题，是回溯算法的典型案例。该问题是国际西洋棋手马克思.贝瑟尔与1848年提出: 在8x8格的国际象棋上摆放八个皇后，使其不能相互攻击，即: 任意两个皇后都不能处于同一行，同一列或同一斜线上，问有多少种摆法？

使用递归法：

import re
num = 0
import numpy as np
import matplotlib.pyplot as plt
def queen_try(i, row_pos, col, left_diag, right_diag):
if i == 8:
    global num
    num = num + 1
    print(row_pos)
    x = np.arange(0,8,1)
    y = row_pos
    plt.grid(b=True,axis='both',color='r',linestyle=':',linewidth=2)
    plt.scatter(x,y,c='k',marker='D')
    plt.show()
    return

for j in range(0,8):
    if col[j] and left_diag[i + j] and right_diag[i - j + 7]:
        row_pos[i] = j
        col[j] = 0
        left_diag[i + j] = 0
        right_diag[i - j + 7] = 0
        queen_try(i + 1, row_pos, col, left_diag, right_diag)
        col[j] = 1
        left_diag[i + j] = 1
        right_diag[i - j + 7] = 1
def main():
    row_pos = [0] * 8
    col = [1] * 8
    left_diag = [1] * 15
    right_diag = [1] * 15
    queen_try(0, row_pos, col, left_diag, right_diag)
main()
print(num)
